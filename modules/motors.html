Сервопривод

Сервопривод - это мотор с обратной связью, способный точно поворачиваться на заданный угол.

#include 

Servo myservo;  // Создаем объект сервопривода
int servoPin = 9;  // Пин для сервопривода

void setup() {
    myservo.attach(servoPin);  // Привязываем сервопривод к пину
}

void loop() {
    // Поворот от 0 до 180 градусов
    for(int angle = 0; angle <= 180; angle += 1) {
        myservo.write(angle);
        delay(15);
    }
    
    // Поворот обратно от 180 до 0 градусов
    for(int angle = 180; angle >= 0; angle -= 1) {
        myservo.write(angle);
        delay(15);
    }
}

DC мотор с драйвером L298N

DC моторы используются для создания движущихся роботов и механизмов. Для управления ими необходим драйвер мотора.

// Пины для управления мотором
const int motorPin1 = 2;  // IN1
const int motorPin2 = 3;  // IN2
const int enablePin = 9;  // ENA

void setup() {
    pinMode(motorPin1, OUTPUT);
    pinMode(motorPin2, OUTPUT);
    pinMode(enablePin, OUTPUT);
}

void forward(int speed) {
    digitalWrite(motorPin1, HIGH);
    digitalWrite(motorPin2, LOW);
    analogWrite(enablePin, speed);
}

void backward(int speed) {
    digitalWrite(motorPin1, LOW);
    digitalWrite(motorPin2, HIGH);
    analogWrite(enablePin, speed);
}

void stop() {
    digitalWrite(motorPin1, LOW);
    digitalWrite(motorPin2, LOW);
    analogWrite(enablePin, 0);
}

void loop() {
    forward(255);   // Полная скорость вперед
    delay(2000);
    stop();         // Остановка
    delay(1000);
    backward(128);  // Половина скорости назад
    delay(2000);
    stop();         // Остановка
    delay(1000);
}

Шаговый двигатель

Шаговые двигатели позволяют осуществлять точное позиционирование и контроль скорости вращения.

#include 

const int stepsPerRevolution = 2048;  // Количество шагов на оборот
const int rolePerMinute = 15;         // Скорость вращения

// Инициализация пинов (IN1, IN3, IN2, IN4)
Stepper myStepper(stepsPerRevolution, 8, 10, 9, 11);

void setup() {
    myStepper.setSpeed(rolePerMinute);
}

void loop() {
    // Полный оборот по часовой стрелке
    myStepper.step(stepsPerRevolution);
    delay(1000);
    
    // Полный оборот против часовой стрелки
    myStepper.step(-stepsPerRevolution);
    delay(1000);
}
