Мигающий светодиод (Blink)

Базовый пример работы со светодиодом


void setup() {
    pinMode(13, OUTPUT);    // Установка пина 13 как выход
}

void loop() {
    digitalWrite(13, HIGH); // Включить светодиод
    delay(1000);           // Ждать 1 секунду
    digitalWrite(13, LOW);  // Выключить светодиод
    delay(1000);           // Ждать 1 секунду
}

Кнопка и светодиод

Управление светодиодом с помощью кнопки


const int buttonPin = 2;    // Пин кнопки
const int ledPin = 13;      // Пин светодиода
int buttonState = 0;        // Состояние кнопки

void setup() {
    pinMode(ledPin, OUTPUT);
    pinMode(buttonPin, INPUT);
}

void loop() {
    buttonState = digitalRead(buttonPin);
    if (buttonState == HIGH) {
        digitalWrite(ledPin, HIGH);
    } else {
        digitalWrite(ledPin, LOW);
    }
}

Аналоговое чтение

Чтение данных с аналогового датчика


const int sensorPin = A0;   // Аналоговый пин датчика
int sensorValue = 0;        // Значение с датчика

void setup() {
    Serial.begin(9600);     // Инициализация Serial порта
}

void loop() {
    sensorValue = analogRead(sensorPin);
    Serial.println(sensorValue);
    delay(100);
}
