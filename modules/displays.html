LCD дисплей (16x2)

LCD дисплей 16x2 - это популярный способ отображения текстовой информации. Он может показывать 16 символов в 2 строках.

#include 

// Инициализация пинов (RS, E, D4, D5, D6, D7)
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
    lcd.begin(16, 2);  // Установка размера дисплея (16 символов, 2 строки)
    lcd.print("Hello, Arduino!");  // Вывод текста
}

void loop() {
    lcd.setCursor(0, 1);  // Перемещение курсора на вторую строку
    lcd.print("Time: ");
    lcd.print(millis() / 1000);  // Вывод времени работы в секундах
    delay(1000);
}

7-сегментный индикатор

7-сегментный индикатор позволяет отображать цифры и некоторые буквы с помощью семи светодиодных сегментов.

// Определение пинов для каждого сегмента (a-g)
int segments[7] = {2, 3, 4, 5, 6, 7, 8};

// Массив для отображения цифр (0-9)
byte numbers[10] = {
    0b1111110,  // 0
    0b0110000,  // 1
    0b1101101,  // 2
    0b1111001,  // 3
    0b0110011,  // 4
    0b1011011,  // 5
    0b1011111,  // 6
    0b1110000,  // 7
    0b1111111,  // 8
    0b1111011   // 9
};

void setup() {
    for (int i = 0; i < 7; i++) {
        pinMode(segments[i], OUTPUT);
    }
}

void displayNumber(int num) {
    if (num >= 0 && num <= 9) {
        for (int i = 0; i < 7; i++) {
            digitalWrite(segments[i], bitRead(numbers[num], i));
        }
    }
}

void loop() {
    // Счет от 0 до 9
    for (int i = 0; i < 10; i++) {
        displayNumber(i);
        delay(1000);
    }
}

OLED дисплей

OLED дисплеи предоставляют возможность отображения графики и текста с высоким разрешением.

#include 
#include 

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

void setup() {
    display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
    display.clearDisplay();
    
    display.setTextSize(1);
    display.setTextColor(WHITE);
    display.setCursor(0,0);
    display.println("OLED Display Test");
    display.display();
}

void loop() {
    // Рисование прямоугольника
    display.clearDisplay();
    display.drawRect(10, 10, 50, 30, WHITE);
    display.display();
    delay(1000);
    
    // Рисование круга
    display.clearDisplay();
    display.drawCircle(64, 32, 20, WHITE);
    display.display();
    delay(1000);
}
